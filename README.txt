const DJNash=(()=>{
  const ADMIN_PASS='nashadmin';
  const defaults={liveDj:'DJ 6PaC',eventName:'Soirée au Nash',djs:[{name:'DJ 6PaC',pass:'nashdj'},{name:'DJ Frank',pass:'frankdj'},{name:'DJ LAMB3RT',pass:'lamb3rtdj'}],requests:[]};
  const load=()=>{const raw=localStorage.getItem('djNashData');return raw?JSON.parse(raw):structuredClone(defaults)};
  const save=(data)=>{localStorage.setItem('djNashData',JSON.stringify(data));window.dispatchEvent(new Event('storage'))};
  const $=id=>document.getElementById(id);
  const esc=s=>String(s||'').replace(/[&<>"]/g,m=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;'}[m]));
  const now=()=>new Date().toLocaleTimeString('fr-CA',{hour:'2-digit',minute:'2-digit'});
  function renderRequests(el,admin=false){const data=load(); if(!el)return; el.innerHTML=data.requests.length?data.requests.slice().reverse().map(r=>`<div class="req ${r.played?'played':''}"><strong>${esc(r.type)}</strong> <small>${esc(r.time)}</small><br><b>${esc(r.name)}</b><p>${esc(r.message)}</p><div class="req-actions"><button onclick="DJNash.togglePlayed('${r.id}')">${r.played?'Remettre':'Jouée ✅'}</button>${admin?`<button class="danger" onclick="DJNash.deleteReq('${r.id}')">Supprimer</button>`:''}</div></div>`).join(''):'<p>Aucune demande pour le moment.</p>';}
  function fillDjs(select){const data=load(); select.innerHTML=data.djs.map(d=>`<option>${esc(d.name)}</option>`).join('');}
  function initPublic(){const data=load();$('publicDj').textContent=data.liveDj;$('publicEvent').textContent=data.eventName;$('requestForm').addEventListener('submit',e=>{e.preventDefault();const data=load();data.requests.push({id:crypto.randomUUID(),time:now(),name:$('name').value,type:$('type').value,message:$('message').value,played:false});save(data);$('requestForm').reset();$('sentMsg').classList.remove('hidden');setTimeout(()=>$('sentMsg').classList.add('hidden'),3000);});setInterval(()=>{const d=load();$('publicDj').textContent=d.liveDj;$('publicEvent').textContent=d.eventName;},1200)}
  function initDJ(){fillDjs($('djSelect'));$('djLogin').addEventListener('submit',e=>{e.preventDefault();const data=load();const dj=data.djs.find(d=>d.name===$('djSelect').value&&d.pass===$('djPass').value);if(!dj){$('djError').classList.remove('hidden');return}data.liveDj=dj.name;save(data);$('loginCard').classList.add('hidden');$('djPanel').classList.remove('hidden');$('djPanelName').textContent=dj.name;$('djPanelEvent').textContent=data.eventName;renderRequests($('requestsList'));setInterval(()=>{const d=load();$('djPanelEvent').textContent=d.eventName;renderRequests($('requestsList'));},1200)});$('clearPlayed').onclick=()=>{const data=load();data.requests=data.requests.filter(r=>!r.played);save(data);renderRequests($('requestsList'));}}
  function initAdmin(){$('adminLogin').addEventListener('submit',e=>{e.preventDefault();if($('adminPass').value!==ADMIN_PASS){$('adminError').classList.remove('hidden');return}$('adminLoginCard').classList.add('hidden');$('adminPanel').classList.remove('hidden');renderAdmin();});}
  function renderAdmin(){const data=load();fillDjs($('liveDj'));$('liveDj').value=data.liveDj;$('eventName').value=data.eventName;$('djAdminList').innerHTML=data.djs.map((d,i)=>`<div class="req"><strong>${esc(d.name)}</strong><br><small>Mot de passe: ${esc(d.pass)}</small><div class="req-actions"><button class="danger" onclick="DJNash.removeDj(${i})">Supprimer</button></div></div>`).join('');renderRequests($('adminRequests'),true);$('liveSettings').onsubmit=e=>{e.preventDefault();const data=load();data.liveDj=$('liveDj').value;data.eventName=$('eventName').value||'Soirée au Nash';save(data);renderAdmin();};$('addDjForm').onsubmit=e=>{e.preventDefault();const data=load();data.djs.push({name:$('newDjName').value,pass:$('newDjPass').value});save(data);$('addDjForm').reset();renderAdmin();};$('clearAll').onclick=()=>{const data=load();data.requests=[];save(data);renderAdmin();};}
  function togglePlayed(id){const data=load();const r=data.requests.find(x=>x.id===id);if(r)r.played=!r.played;save(data);renderRequests($('requestsList'));renderRequests($('adminRequests'),true)}
  function deleteReq(id){const data=load();data.requests=data.requests.filter(r=>r.id!==id);save(data);renderAdmin()}
  function removeDj(i){const data=load();data.djs.splice(i,1);save(data);renderAdmin()}
  return{initPublic,initDJ,initAdmin,togglePlayed,deleteReq,removeDj};
})();
